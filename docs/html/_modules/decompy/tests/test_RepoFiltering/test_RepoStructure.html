        meta_files = [config.open().read() for config in Path(self.rs.root).glob("*/Config.META")]

        def correct_meta(file):
            j = json.loads(file)
            self.assertTrue({"url", "name", "author", "filter-date"}.issubset(set(j.keys())))

        for file in meta_files:
            correct_meta(file)
